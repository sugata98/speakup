<ul>

  <li>CSS 3</li>
  <li>JavaScript</li>
  <li>Bootstrap 4</li>
  <li>jQuery</li>
  <li>Node JS</li>
  <li>EXpress JS</li>
  <li>MongoDB</li>
  <li><em>Authentication</em> using PassportJS</li>
</ul>
